// Procedural Tex, Fence Mesh - Source: https://www.shadertoy.com/view/4styR4 
const float PI = 3.14159265359;
const float ZOOM = 20.0;
const float WEIGHT = 3.5;

vec4 main(vec2 fragCoord)
{ 
    vec2 uv = ZOOM * PI * fragCoord/iResolution.xy
            * vec2(iResolution.x/iResolution.y, 1.);
    float d = sin(uv.x)+cos(uv.y);
  	float distance = d * WEIGHT;
    vec3 color = vec3(abs(distance));
  	return vec4(color, 1.0); 
}
